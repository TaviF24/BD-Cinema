create table Regizori(
	regizor_id number(5) constraint pk_regizor primary key,
	nume_regizor varchar2(30) constraint null_nume_regizor not null,
	prenume_regizor varchar2(30) constraint null_prenume_regizor not null
);

create table Filme(
	film_id number(5) constraint pk_film primary key,
	nume_film varchar2(30) constraint null_nume_film not null,
    regizor_id number(5) constraint null_regizor_filme not null,
	rating number(2) constraint null_rating_film not null,
    gen varchar2(10) constraint null_gen_film not null,
    data_lansare date constraint null_data_film not null,
    durata number(3) constraint null_durata_film not null,
    constraint fk_regizor_filme foreign key (regizor_id) references regizori(regizor_id) on delete cascade,
    constraint check_rating_film check(1<=rating and rating<=10),
    constraint check_gen_film check(lower(gen) in ('actiune','drama','horror','thriller','comedie','aventura','romantic','mister'))
);

create table Clienti(
	client_id number(5) constraint pk_client primary key,
	nume_client varchar2(30) constraint null_nume_client not null,
    prenume_client varchar2(40),
    telefon varchar2(20) constraint unq_telefon_client unique,
    email varchar2(30) constraint null_email_client not null,
    constraint unq_email_client unique(email) 
);   

create table Snacks_uri(
	snack_id number(5) constraint pk_snack primary key,
	nume_snack varchar2(30) constraint null_nume_snack not null,
	pret number(4,2) constraint null_pret_snack not null,
    constraint check_pret_snack check(pret>=0 and pret<=60)
);

create table Tranzactii(
    client_id number(5) constraint fk_client_tranzactii references clienti(client_id) on delete cascade,
	snack_id number(5) constraint fk_snack_tranzactii references snacks_uri(snack_id) on delete cascade,
	data date default sysdate,
    constraint pk_tranzactie primary key(client_id,snack_id) );

create table Sectiuni_lucru(
    sectiune_id number(5) constraint pk_sectiune primary key,
    nume_sectiune varchar2(30) constraint null_nume_sectiuni not null,
    constraint unq_nume_sectiune unique(nume_sectiune)
);

create table Locatii(
	locatie_id number(5) constraint pk_locatie primary key,
	strada varchar2(30) constraint null_strada_locatie not null,
    localitate varchar2(30) constraint null_localitate_locatie not null,
    judet varchar2(30) constraint null_judet_locatie not null,
    constraint unq_StrLocJud unique(strada,localitate,judet)
);


create table Cinematografe(
    cinematograf_id number(5) constraint pk_cinematograf primary key,
    locatie_id number(5) constraint null_locatie_cinematografe not null,
	numar_sali number(2) constraint null_numar_sali not null,
    constraint fk_locatie_cinematografe foreign key (locatie_id) references locatii(locatie_id) on delete cascade,
    constraint check_nrsali_cinematograf check(numar_sali>=1) );  

create table Ture_angajati(
	tura_id number(5) constraint pk_tura primary key,
	cinematograf_id number(5) constraint null_cinematograf_ture not null,
    ora_incepere varchar2(30) default to_char(sysdate,'hh24:mi'),
    durata number(1) constraint null_durata_ture not null,
    constraint fk_cinematograf_ture foreign key (cinematograf_id) references cinematografe(cinematograf_id) on delete cascade,
    constraint check_durata_ture check(durata in (4,5,6,7,8))
);



create table Angajati(
	angajat_id number(5) constraint pk_angajat primary key,
    sectiune_id number(5) constraint null_sectiune_angajati not null,
    tura_id number(5) constraint null_tura_angajati not null,
    nume_angajat varchar2(30) constraint null_nume_angajati not null,
    prenume_angajat varchar2(40) constraint null_prenume_angajati not null,
    salariu number(6,2) constraint null_salariu_angajati not null,
    telefon varchar2(20) constraint null_telefon_angajati not null,
    constraint fk_sectiune_angajati foreign key (sectiune_id) references sectiuni_lucru(sectiune_id) on delete cascade,
    constraint fk_tura_angajati foreign key (tura_id) references ture_angajati(tura_id) on delete cascade,
    constraint check_salariu_angajati check(1500<=salariu and salariu<=3000),
    constraint unq_telefon_angajati unique(telefon)
);

create table Sali_cinematograf(
    sala_id number(5) constraint pk_sala primary key,
    cinematograf_id number(5) constraint null_cinematograf_sali not null,
	capacitate number(3) constraint null_capacitate_sali not null,
    constraint fk_cinematograf_sali foreign key (cinematograf_id) references cinematografe(cinematograf_id) on delete cascade,
    constraint check_capacitate_sali check(1<=capacitate and capacitate<=150)
); 

create table Transmiteri(
    transmitere_id number(5) constraint pk_transmitere primary key,
    film_id number(5) constraint null_film_transmiteri not null,
	sala_id number(5) constraint null_sala_transmiteri not null,
    data_ora_incepere varchar2(50) default to_char(sysdate,'dd-mm-yyyy, hh24:mi'),
    pret number(5,2) constraint null_pret_transmiteri not null,
    constraint fk_film_transmiteri foreign key (film_id) references filme(film_id) on delete cascade,
    constraint fk_sala_transmiteri foreign key (sala_id) references sali_cinematograf(sala_id) on delete cascade,
    constraint check_pret_transmiteri check(0<=pret and pret<=100),
    constraint unq_film_sala_data unique(film_id,sala_id, data_ora_incepere)
);     

create table Locuri_sala(
    loc_id number(5) constraint pk_loc primary key,
    sala_id number(5) constraint null_sala_locuri not null,
	litera_rand varchar2(1) constraint null_litera_locuri not null,
    numar_loc number(3) constraint null_numar_locuri not null,
    constraint fk_sala_locuri foreign key (sala_id) references sali_cinematograf(sala_id) on delete cascade,
    constraint unq_sala_litera_numar unique(sala_id,litera_rand,numar_loc)
); 

create table Bilete(
    bilet_id number(5) constraint pk_bilet primary key,
    loc_id number(5) constraint null_loc_bilete not null,
    constraint fk_loc_bilete foreign key (loc_id) references locuri_sala(loc_id) on delete cascade,
    constraint unq_loc_bilete unique(loc_id)
);


create table Rezervari(
    client_id number(5) constraint fk_client_rezervari references clienti(client_id) on delete cascade,
	transmitere_id number(5) constraint fk_transmitere_rezervari references transmiteri(transmitere_id) on delete cascade,
    bilet_id number(5) constraint fk_bilet_rezervari references bilete(bilet_id) on delete cascade,
    constraint pk_rezervare primary key(client_id,transmitere_id,bilet_id),
    constraint unq_transmitere_bilet unique(transmitere_id,bilet_id)
);
